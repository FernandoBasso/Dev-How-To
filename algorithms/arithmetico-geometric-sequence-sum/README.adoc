[source,ruby,lineos]
----
def sum_terms(n)
  # This uses `map` and `inject`. It will "loop" twice.
  # (1..n).map { |i| i ** 2 + 1}.inject(0, &:+)

  # This uses only `inject`. It will loop only once.
  (1..n).inject(0) { |acc, i| acc += i ** 2 + 1 }
end
----
